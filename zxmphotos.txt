    <script>
        let photos = [];
        let currentPhotoId = null;

        function toggleUpload(show) {
            document.getElementById('upload-modal').classList.toggle('hidden', !show);
        }

        document.getElementById('file-input').addEventListener('change', function(e) {
            const files = Array.from(e.target.files);
            if (files.length > 15) return alert('æœ€å¤šä¸Šä¼ 15å¼ ');
            
            files.forEach(file => {
                const reader = new FileReader();
                reader.onload = function(event) {
                    const photo = {
                        id: Date.now() + Math.random(),
                        url: event.target.result,
                        likes: 0,
                        hasLiked: false,
                        comments: []
                    };
                    photos.unshift(photo);
                    renderGrid();
                };
                reader.readAsDataURL(file);
            });
            toggleUpload(false);
        });

        function renderGrid() {
            const grid = document.getElementById('photo-grid');
            const empty = document.getElementById('empty-state');
            
            if (photos.length > 0) {
                grid.classList.remove('hidden');
                empty.classList.add('hidden');
                grid.innerHTML = photos.map(p => `
                    <div onclick="openDetail(${p.id})" class="photo-card group relative aspect-[3/4] rounded-[2rem] overflow-hidden cursor-pointer bg-white shadow-sm border border-white">
                        <img src="${p.url}" class="w-full h-full object-cover">
                        <div class="absolute inset-0 bg-black/20 opacity-0 group-hover:opacity-100 transition-opacity flex items-end p-6">
                            <div class="flex gap-4 text-white font-bold text-sm">
                                <span class="flex items-center gap-1">â¤ ${p.likes}</span>
                                <span class="flex items-center gap-1">ğŸ’¬ ${p.comments.length}</span>
                            </div>
                        </div>
                    </div>
                `).join('');
            }
        }

        function openDetail(id) {
            currentPhotoId = id;
            const photo = photos.find(p => p.id === id);
            document.getElementById('detail-img').src = photo.url;
            document.getElementById('like-count').innerText = photo.likes;
            
            const likeBtn = document.getElementById('like-btn');
            const heart = document.getElementById('heart-icon');
            
            if(photo.hasLiked) {
                likeBtn.className = "flex items-center gap-2 px-6 py-2.5 rounded-2xl transition-all font-bold text-sm bg-red-500 text-white";
                heart.setAttribute('fill', 'white');
            } else {
                likeBtn.className = "flex items-center gap-2 px-6 py-2.5 rounded-2xl transition-all font-bold text-sm bg-slate-100 text-slate-400";
                heart.setAttribute('fill', 'none');
                likeBtn.onclick = () => {
                    photo.likes++;
                    photo.hasLiked = true;
                    openDetail(id); // åˆ·æ–°æ˜¾ç¤º
                    renderGrid();
                };
            }
            
            document.getElementById('comment-list').innerHTML = photo.comments.map(c => `
                <div class="bg-slate-50 p-4 rounded-2xl">
                    <p class="text-[10px] font-black text-[#9575CD] mb-1 uppercase">${c.user}</p>
                    <p class="text-sm text-slate-700">${c.text}</p>
                </div>
            `).join('') || '<p class="text-center text-slate-300 py-10 italic text-sm">æš‚æ— è¯„è®º</p>';
            
            document.getElementById('detail-modal').classList.remove('hidden');
        }

        function closeDetail() {
            document.getElementById('detail-modal').classList.add('hidden');
        }

        function submitComment() {
            const input = document.getElementById('comment-input');
            if (!input.value.trim()) return;
            
            const photo = photos.find(p => p.id === currentPhotoId);
            photo.comments.push({ user: 'è®¿å®¢', text: input.value });
            input.value = '';
            openDetail(currentPhotoId);
            renderGrid();
        }
    </script>
